124static inline void in_mul(Unit *out, const Unit *x, size_t n, Unit y)
125{
126 assert(n > 0);
127 Unit H = 0;
128 for (size_t i = 0; i < n; i++) {
129 Unit t = H;
130 Unit L = mulUnit(&H, x[i], y);
131 out[i] = t + L;
132 if (out[i] < t) {
133 H++;
134 }
135 }
136 out[n] = H;
137}

67 class ExceptionMatcher : public Catch::MatcherBase<SpecialException> {
68 int m_expected;
69 public:
70 ExceptionMatcher(int i) : m_expected(i) {}

72 bool match(SpecialException const &se) const override {
73 return se.i == m_expected;
74 }
75

77 std::ostringstream ss;

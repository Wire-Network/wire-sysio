43 // Encapsulates a name ($whatever) parsed from the WAST file.
44 // References the characters in the input string the name was parsed from,
45 // so it can handle arbitrary length names in a fixed size struct, but only
46 // as long as the input string isn't freed.
47 // Includes a hash of the name's characters.
48 struct Name
49 {
50 Name(): begin(nullptr), numChars(0), hash(0) {}
51 Name(const char* inBegin,U32 inNumChars)
52 : begin(inBegin)
53 , numChars(inNumChars)
54 , hash(calcHash(inBegin,inNumChars))
55 {}
56
57 operator bool() const { return begin != nullptr; }
58 std::string getString() const { return begin ? std::string(begin + 1,numChars - 1) : std::string(); }
59 Uptr getCharOffset(const char* string) const { return begin - string; }
60
61 friend bool operator==(const Name& a,const Name& b)
62 {
63 return a.hash == b.hash && a.numChars == b.numChars && memcmp(a.begin,b.begin,a.numChars) == 0;
64 }
65 friend bool operator!=(const Name& a,const Name& b)
66 {
67 return !(a == b);
68 }

185 // Parses the surrounding parentheses for an inner parser, and handles recovery at the closing parenthesis.
186 template<typename ParseInner>
187 static void parseParenthesized(ParseState& state,ParseInner parseInner)
188 {
189 const Token* openingParenthesisToken = state.nextToken;
190 require(state,t_leftParenthesis);
191 try
192 {
193 parseInner();
194 require(state,t_rightParenthesis);
195 }
196 catch(RecoverParseException)
197 {
198 findClosingParenthesis(state,openingParenthesisToken);
199 }
200 }

202 // Tries to parse '(' tagType parseInner ')', handling recovery at the closing parenthesis.
203 // Returns true if any tokens were consumed.
204 template<typename ParseInner>
205 static bool tryParseParenthesizedTagged(ParseState& state,TokenType tagType,ParseInner parseInner)
206 {
207 const Token* openingParenthesisToken = state.nextToken;
208 if(state.nextToken[0].type != t_leftParenthesis || state.nextToken[1].type != tagType)
209 {
210 return false;
211 }
212 try
213 {
214 state.nextToken += 2;
215 parseInner();
216 require(state,t_rightParenthesis);
217 }
218 catch(RecoverParseException)
219 {
220 findClosingParenthesis(state,openingParenthesisToken);
221 }
222 return true;
223 }

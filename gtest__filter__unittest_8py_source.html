31"""Unit test for Google Test test filters.
32
33A user can specify which test(s) in a Google Test program to run via either
34the GTEST_FILTER environment variable or the --gtest_filter flag.
35This script tests such functionality by invoking
36gtest_filter_unittest_ (a program written with Google Test) with different
37environments and command line flags.
38
39Note that test sharding may also influence which tests are filtered. Therefore,
40we test that here also.

53# Checks if this platform can pass empty environment variables to child
54# processes. We set an env variable to an empty string and invoke a python
55# script in a subprocess to print whether the variable is STILL in
56# os.environ. We then use 'eval' to parse the child's output so that an
57# exception is thrown if the input is anything other than 'True' nor 'False'.
58CAN_PASS_EMPTY_ENV = False
59if sys.executable:
60 os.environ['EMPTY_VAR'] = ''
61 child = gtest_test_utils.Subprocess(
62 [sys.executable, '-c', 'import os; print \'EMPTY_VAR\' in os.environ'])
63 CAN_PASS_EMPTY_ENV = eval(child.output)
64
65
66# Check if this platform can unset environment variables in child processes.
67# We set an env variable to a non-empty string, unset it, and invoke
68# a python script in a subprocess to print whether the variable
69# is NO LONGER in os.environ.
70# We use 'eval' to parse the child's output so that an exception
71# is thrown if the input is neither 'True' nor 'False'.
72CAN_UNSET_ENV = False
73if sys.executable:
74 os.environ['UNSET_VAR'] = 'X'
75 del os.environ['UNSET_VAR']
76 child = gtest_test_utils.Subprocess(
77 [sys.executable, '-c', 'import os; print \'UNSET_VAR\' not in os.environ'
78 ])
79 CAN_UNSET_ENV = eval(child.output)

173def SetEnvVar(env_var, value):
174 """Sets the env variable to 'value'; unsets it when 'value' is None."""
175
176 if value is not None:
177 environ[env_var] = value
178 elif env_var in environ:
179 del environ[env_var]

182def RunAndReturnOutput(args = None):
183 """Runs the test program and returns its output."""
184
185 return gtest_test_utils.Subprocess([COMMAND] + (args or []),
186 env=environ).output

189def RunAndExtractTestList(args = None):
190 """Runs the test program and returns its exit code and a list of tests run."""
191
192 p = gtest_test_utils.Subprocess([COMMAND] + (args or []), env=environ)
193 tests_run = []
194 test_case = ''
195 test = ''
196 for line in p.output.split('\n'):
197 match = TEST_CASE_REGEX.match(line)
198 if match is not None:
199 test_case = match.group(1)
200 else:
201 match = TEST_REGEX.match(line)
202 if match is not None:
203 test = match.group(1)
204 tests_run.append(test_case + '.' + test)
205 return (tests_run, p.exit_code)

208def InvokeWithModifiedEnv(extra_env, function, *args, **kwargs):
209 """Runs the given function and arguments in a modified environment."""
210 try:
211 original_env = environ.copy()
212 environ.update(extra_env)
213 return function(*args, **kwargs)
214 finally:
215 environ.clear()
216 environ.update(original_env)

252 def AdjustForParameterizedTests(self, tests_to_run):
253 """Adjust tests_to_run in case value parameterized tests are disabled."""
254
255 global param_tests_present
256 if not param_tests_present:
257 return list(sets.Set(tests_to_run) - sets.Set(PARAM_TESTS))

259 return tests_to_run

261 def RunAndVerify(self, gtest_filter, tests_to_run):
262 """Checks that the binary runs correct set of tests for a given filter."""
263
264 tests_to_run = self.AdjustForParameterizedTests(tests_to_run)
265
266 # First, tests using the environment variable.
267
268 # Windows removes empty variables from the environment when passing it
269 # to a new process. This means it is impossible to pass an empty filter
270 # into a process using the environment variable. However, we can still
271 # test the case when the variable is not supplied (i.e., gtest_filter is
272 # None).
273 # pylint: disable-msg=C6403
274 if CAN_TEST_EMPTY_FILTER or gtest_filter != '':
275 SetEnvVar(FILTER_ENV_VAR, gtest_filter)
276 tests_run = RunAndExtractTestList()[0]
277 SetEnvVar(FILTER_ENV_VAR, None)
278 self.AssertSetEqual(tests_run, tests_to_run)
279 # pylint: enable-msg=C6403
280
281 # Next, tests using the command line flag.
282
283 if gtest_filter is None:
284 args = []
285 else:
286 args = ['--%s=%s' % (FILTER_FLAG, gtest_filter)]
287
288 tests_run = RunAndExtractTestList(args)[0]
289 self.AssertSetEqual(tests_run, tests_to_run)

291 def RunAndVerifyWithSharding(self, gtest_filter, total_shards, tests_to_run,
292 args=None, check_exit_0=False):
293 """Checks that binary runs correct tests for the given filter and shard.
294
295 Runs all shards of gtest_filter_unittest_ with the given filter, and
296 verifies that the right set of tests were run. The union of tests run
297 on each shard should be identical to tests_to_run, without duplicates.
298 If check_exit_0, .
299
300 Args:
301 gtest_filter: A filter to apply to the tests.
302 total_shards: A total number of shards to split test run into.
303 tests_to_run: A set of tests expected to run.
304 args : Arguments to pass to the to the test binary.
305 check_exit_0: When set to a true value, make sure that all shards
306 return 0.
307 """
308
309 tests_to_run = self.AdjustForParameterizedTests(tests_to_run)
310
311 # Windows removes empty variables from the environment when passing it
312 # to a new process. This means it is impossible to pass an empty filter
313 # into a process using the environment variable. However, we can still
314 # test the case when the variable is not supplied (i.e., gtest_filter is
315 # None).
316 # pylint: disable-msg=C6403
317 if CAN_TEST_EMPTY_FILTER or gtest_filter != '':
318 SetEnvVar(FILTER_ENV_VAR, gtest_filter)
319 partition = []
320 for i in range(0, total_shards):
321 (tests_run, exit_code) = RunWithSharding(total_shards, i, args)
322 if check_exit_0:
323 self.assertEqual(0, exit_code)
324 partition.append(tests_run)
325
326 self.AssertPartitionIsValid(tests_to_run, partition)
327 SetEnvVar(FILTER_ENV_VAR, None)
328 # pylint: enable-msg=C6403
329
330 def RunAndVerifyAllowingDisabled(self, gtest_filter, tests_to_run):
331 """Checks that the binary runs correct set of tests for the given filter.
332
333 Runs gtest_filter_unittest_ with the given filter, and enables
334 disabled tests. Verifies that the right set of tests were run.
335
336 Args:
337 gtest_filter: A filter to apply to the tests.
338 tests_to_run: A set of tests expected to run.
339 """
340
341 tests_to_run = self.AdjustForParameterizedTests(tests_to_run)
342
343 # Construct the command line.
344 args = ['--%s' % ALSO_RUN_DISABLED_TESTS_FLAG]
345 if gtest_filter is not None:
346 args.append('--%s=%s' % (FILTER_FLAG, gtest_filter))
347
348 tests_run = RunAndExtractTestList(args)[0]
349 self.AssertSetEqual(tests_run, tests_to_run)

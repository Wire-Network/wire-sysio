70uint16_t yubikey_crc16(const uint8_t *buf, size_t buf_size) {
71 uint16_t m_crc = 0xffff;
72
73 while (buf_size--) {
74 int i, j;
75 m_crc ^= (uint8_t) *buf++ & 0xFF;
76 for (i = 0; i < 8; i++) {
77 j = m_crc & 1;
78 m_crc >>= 1;
79 if (j) {
80 m_crc ^= 0x8408;
81 }
82 }
83 }
84
85 return m_crc;
86}

138mnt4_G1 mnt4_G1::operator+(const mnt4_G1 &other) const
139{
140 // handle special cases having to do with O
141 if (this->is_zero())
142 {
143 return other;
144 }
145
146 if (other.is_zero())
147 {
148 return *this;
149 }
150
151 // no need to handle points of order 2,4
152 // (they cannot exist in a prime-order subgroup)
153
154 // handle double case, and then all the rest
155 /*
156 The code below is equivalent to (but faster than) the snippet below:
157
158 if (this->operator==(other))
159 {
160 return this->dbl();
161 }
162 else
163 {
164 return this->add(other);
165 }
166 */
167
168 const mnt4_Fq X1Z2 = (this->X) * (other.Z); // X1Z2 = X1*Z2
169 const mnt4_Fq X2Z1 = (this->Z) * (other.X); // X2Z1 = X2*Z1
170
171 // (used both in add and double checks)
172
173 const mnt4_Fq Y1Z2 = (this->Y) * (other.Z); // Y1Z2 = Y1*Z2
174 const mnt4_Fq Y2Z1 = (this->Z) * (other.Y); // Y2Z1 = Y2*Z1
175
176 if (X1Z2 == X2Z1 && Y1Z2 == Y2Z1)
177 {
178 // perform dbl case
179 const mnt4_Fq XX = (this->X).squared(); // XX = X1^2
180 const mnt4_Fq ZZ = (this->Z).squared(); // ZZ = Z1^2
181 const mnt4_Fq w = mnt4_G1::coeff_a * ZZ + (XX + XX + XX); // w = a*ZZ + 3*XX
182 const mnt4_Fq Y1Z1 = (this->Y) * (this->Z);
183 const mnt4_Fq s = Y1Z1 + Y1Z1; // s = 2*Y1*Z1
184 const mnt4_Fq ss = s.squared(); // ss = s^2
185 const mnt4_Fq sss = s * ss; // sss = s*ss
186 const mnt4_Fq R = (this->Y) * s; // R = Y1*s
187 const mnt4_Fq RR = R.squared(); // RR = R^2
188 const mnt4_Fq B = ((this->X)+R).squared()-XX-RR; // B = (X1+R)^2 - XX - RR
189 const mnt4_Fq h = w.squared() - (B+B); // h = w^2 - 2*B
190 const mnt4_Fq X3 = h * s; // X3 = h*s
191 const mnt4_Fq Y3 = w * (B-h)-(RR+RR); // Y3 = w*(B-h) - 2*RR
192 const mnt4_Fq Z3 = sss; // Z3 = sss
193
194 return mnt4_G1(X3, Y3, Z3);
195 }
196
197 // if we have arrived here we are in the add case
198 const mnt4_Fq Z1Z2 = (this->Z) * (other.Z); // Z1Z2 = Z1*Z2
199 const mnt4_Fq u = Y2Z1 - Y1Z2; // u = Y2*Z1-Y1Z2
200 const mnt4_Fq uu = u.squared(); // uu = u^2
201 const mnt4_Fq v = X2Z1 - X1Z2; // v = X2*Z1-X1Z2
202 const mnt4_Fq vv = v.squared(); // vv = v^2
203 const mnt4_Fq vvv = v * vv; // vvv = v*vv
204 const mnt4_Fq R = vv * X1Z2; // R = vv*X1Z2
205 const mnt4_Fq A = uu * Z1Z2 - (vvv + R + R); // A = uu*Z1Z2 - vvv - 2*R
206 const mnt4_Fq X3 = v * A; // X3 = v*A
207 const mnt4_Fq Y3 = u * (R-A) - vvv * Y1Z2; // Y3 = u*(R-A) - vvv*Y1Z2
208 const mnt4_Fq Z3 = vvv * Z1Z2; // Z3 = vvv*Z1Z2
209
210 return mnt4_G1(X3, Y3, Z3);
211}

225{
226 // handle special cases having to do with O
227 if (this->is_zero())
228 {
229 return other;
230 }
231
232 if (other.is_zero())
233 {
234 return (*this);
235 }
236
237 // no need to handle points of order 2,4
238 // (they cannot exist in a prime-order subgroup)
239
240 // handle double case
241 if (this->operator==(other))
242 {
243 return this->dbl();
244 }
245
246#ifdef PROFILE_OP_COUNTS

248#endif

254 const mnt4_Fq Z1Z2 = (this->Z) * (other.Z); // Z1Z2 = Z1*Z2

256 const mnt4_Fq uu = u.squared(); // uu = u^2
257 const mnt4_Fq v = (other.X) * (this->Z) - X1Z2; // v = X2*Z1-X1Z2
258 const mnt4_Fq vv = v.squared(); // vv = v^2
259 const mnt4_Fq vvv = v * vv; // vvv = v*vv
260 const mnt4_Fq R = vv * X1Z2; // R = vv*X1Z2
261 const mnt4_Fq A = uu * Z1Z2 - (vvv + R + R); // A = uu*Z1Z2 - vvv - 2*R
262 const mnt4_Fq X3 = v * A; // X3 = v*A
263 const mnt4_Fq Y3 = u * (R-A) - vvv * Y1Z2; // Y3 = u*(R-A) - vvv*Y1Z2
264 const mnt4_Fq Z3 = vvv * Z1Z2; // Z3 = vvv*Z1Z2

269mnt4_G1 mnt4_G1::mixed_add(const mnt4_G1 &other) const
270{
271#ifdef PROFILE_OP_COUNTS
272 this->add_cnt++;
273#endif
274 // NOTE: does not handle O and pts of order 2,4
275 // http://www.hyperelliptic.org/EFD/g1p/auto-shortw-projective.html#addition-add-1998-cmo-2
276 //assert(other.Z == mnt4_Fq::one());
277
278 if (this->is_zero())
279 {
280 return other;
281 }
282
283 if (other.is_zero())
284 {
285 return (*this);
286 }
287
288#ifdef DEBUG
289 assert(other.is_special());
290#endif
291
292 const mnt4_Fq &X1Z2 = (this->X); // X1Z2 = X1*Z2 (but other is special and not zero)
293 const mnt4_Fq X2Z1 = (this->Z) * (other.X); // X2Z1 = X2*Z1
294
295 // (used both in add and double checks)
296
297 const mnt4_Fq &Y1Z2 = (this->Y); // Y1Z2 = Y1*Z2 (but other is special and not zero)
298 const mnt4_Fq Y2Z1 = (this->Z) * (other.Y); // Y2Z1 = Y2*Z1
299
300 if (X1Z2 == X2Z1 && Y1Z2 == Y2Z1)
301 {
302 return this->dbl();
303 }
304
305 const mnt4_Fq u = Y2Z1 - this->Y; // u = Y2*Z1-Y1
306 const mnt4_Fq uu = u.squared(); // uu = u2
307 const mnt4_Fq v = X2Z1 - this->X; // v = X2*Z1-X1
308 const mnt4_Fq vv = v.squared(); // vv = v2
309 const mnt4_Fq vvv = v*vv; // vvv = v*vv
310 const mnt4_Fq R = vv * this->X; // R = vv*X1
311 const mnt4_Fq A = uu * this->Z - vvv - R - R; // A = uu*Z1-vvv-2*R
312 const mnt4_Fq X3 = v * A; // X3 = v*A
313 const mnt4_Fq Y3 = u*(R-A) - vvv * this->Y; // Y3 = u*(R-A)-vvv*Y1
314 const mnt4_Fq Z3 = vvv * this->Z; // Z3 = vvv*Z1
315
316 return mnt4_G1(X3, Y3, Z3);

352bool mnt4_G1::is_well_formed() const
353{
354 if (this->is_zero())
355 {
356 return true;
357 }
358 else
359 {
360 /*
361 y^2 = x^3 + ax + b
362
363 We are using projective, so equation we need to check is actually
364
365 (y/z)^2 = (x/z)^3 + a (x/z) + b
366 z y^2 = x^3 + a z^2 x + b z^3
367
368 z (y^2 - b z^2) = x ( x^2 + a z^2)
369 */
370 const mnt4_Fq X2 = this->X.squared();
371 const mnt4_Fq Y2 = this->Y.squared();
372 const mnt4_Fq Z2 = this->Z.squared();
373
374 return (this->Z * (Y2 - mnt4_G1::coeff_b * Z2) == this->X * (X2 + mnt4_G1::coeff_a * Z2));
375 }

409std::istream& operator>>(std::istream &in, mnt4_G1 &g)
410{
411 char is_zero;
412 mnt4_Fq tX, tY;
413
414#ifdef NO_PT_COMPRESSION
415 in >> is_zero >> tX >> tY;
416 is_zero -= '0';
417#else
418 in.read((char*)&is_zero, 1);
419 is_zero -= '0';
420 consume_OUTPUT_SEPARATOR(in);
421
422 unsigned char Y_lsb;
423 in >> tX;
424 consume_OUTPUT_SEPARATOR(in);
425 in.read((char*)&Y_lsb, 1);
426 Y_lsb -= '0';
427
428 // y = +/- sqrt(x^3 + a*x + b)
429 if (!is_zero)
430 {
431 mnt4_Fq tX2 = tX.squared();
432 mnt4_Fq tY2 = (tX2 + mnt4_G1::coeff_a) * tX + mnt4_G1::coeff_b;
433 tY = tY2.sqrt();
434
435 if ((tY.as_bigint().data[0] & 1) != Y_lsb)
436 {
437 tY = -tY;
438 }
439 }
440#endif
441 // using projective coordinates
442 if (!is_zero)
443 {
444 g.X = tX;
445 g.Y = tY;
446 g.Z = mnt4_Fq::one();
447 }
448 else
449 {
450 g = mnt4_G1::zero();
451 }
452
453 return in;
454}

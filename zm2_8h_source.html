165 static inline void inv(Fp& out, const Fp& x)
166 {
167#ifdef MIE_USE_X64ASM
168 Fp r;
169 int k = preInv(r, x);
170#else
171 static const Fp p(&p_[0]);
172 Fp u, v, r, s;
173 u = p;
174 v = x;
175 r.clear();
176 s.clear(); s[0] = 1; // s is real 1
177 int k = 0;
178 while (!v.isZero()) {
179 if ((u[0] & 1) == 0) {
180 shr1(u, u);
181 addNC(s, s, s);
182 } else if ((v[0] & 1) == 0) {
183 shr1(v, v);
184 addNC(r, r, r);
185 } else if (v >= u) {
186 subNC(v, v, u);
187 addNC(s, s, r);
188 shr1(v, v);
189 addNC(r, r, r);
190 } else {
191 subNC(u, u, v);
192 addNC(r, r, s);
193 shr1(u, u);
194 addNC(s, s, s);
195 }
196 k++;
197 }
198 if (r >= p) {
199 subNC(r, r, p);
200 }
201 assert(!r.isZero());
202 subNC(r, p, r);
203#endif
204 /*
205 xr = 2^k
206 R = 2^256
207 get r2^(-k)R^2 = r 2^(512 - k)

27 static inline constexpr void* align_address(void* addr, size_t align_amt) {
28 if constexpr (should_align_memory_ops) {
29 addr = (void*)(((uintptr_t)addr + (1 << align_amt) - 1) & ~((1 << align_amt) - 1));
30 return addr;
31 } else {
32 return addr;
33 }

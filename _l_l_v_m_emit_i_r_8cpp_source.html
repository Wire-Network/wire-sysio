308 // Bounds checks and converts a memory operation I32 address operand to a LLVM pointer.
309 llvm::Value* coerceByteIndexToPointer(llvm::Value* byteIndex,U32 offset,llvm::Type* memoryType)
310 {
311
312 // On a 64 bit runtime, if the address is 32-bits, zext it to 64-bits.
313 // This is crucial for security, as LLVM will otherwise implicitly sign extend it to 64-bits in the GEP below,
314 // interpreting it as a signed offset and allowing access to memory outside the sandboxed memory range.
315 // There are no 'far addresses' in a 32 bit runtime.
316 byteIndex = irBuilder.CreateZExt(byteIndex,llvmI64Type);
317
318 // Add the offset to the byte index.
319 if(offset)
320 {
321 byteIndex = irBuilder.CreateAdd(byteIndex,irBuilder.CreateZExt(emitLiteral(offset),llvmI64Type));
322 }
323
324 // Cast the pointer to the appropriate type.
325 auto bytePointer = CreateInBoundsGEPWAR(irBuilder, moduleContext.defaultMemoryBase, byteIndex);
326
327 return irBuilder.CreatePointerCast(bytePointer,memoryType->getPointerTo(256));
328 }

373 // A helper function to emit a conditional call to a non-returning intrinsic function.
374 void emitConditionalTrapIntrinsic(llvm::Value* booleanCondition,const char* intrinsicName,const FunctionType* intrinsicType,const std::initializer_list<llvm::Value*>& args)
375 {

384
385 irBuilder.SetInsertPoint(endBlock);

418 void block(ControlStructureImm imm)

420 // Create an end block+phi for the block result.
421 auto endBlock = llvm::BasicBlock::Create(context,"blockEnd",llvmFunction);

430 void loop(ControlStructureImm imm)
431 {
432 // Create a loop block, and an end block+phi for the loop result.
433 auto loopBodyBlock = llvm::BasicBlock::Create(context,"loopBody",llvmFunction);
434 auto endBlock = llvm::BasicBlock::Create(context,"loopEnd",llvmFunction);
435 auto endPHI = createPHI(endBlock,imm.resultType);
436
437 // Branch to the loop body and switch the IR builder to emit there.
438 irBuilder.CreateBr(loopBodyBlock);
439 irBuilder.SetInsertPoint(loopBodyBlock);
440
441 // Push a control context that ends at the end block/phi.
442 pushControlStack(ControlContext::Type::loop,imm.resultType,endBlock,endPHI);
443
444 // Push a branch target for the loop body start.
445 pushBranchTarget(ResultType::none,loopBodyBlock,nullptr);
446 }
447 void if_(ControlStructureImm imm)
448 {
449 // Create a then block and else block for the if, and an end block+phi for the if result.
450 auto thenBlock = llvm::BasicBlock::Create(context,"ifThen",llvmFunction);
451 auto elseBlock = llvm::BasicBlock::Create(context,"ifElse",llvmFunction);
452 auto endBlock = llvm::BasicBlock::Create(context,"ifElseEnd",llvmFunction);
453 auto endPHI = createPHI(endBlock,imm.resultType);
454
455 // Pop the if condition from the operand stack.
456 auto condition = pop();
457 irBuilder.CreateCondBr(coerceI32ToBool(condition),thenBlock,elseBlock);
458
459 // Switch the IR builder to emit the then block.
460 irBuilder.SetInsertPoint(thenBlock);
461
462 // Push an ifThen control context that ultimately ends at the end block/phi, but may
463 // be terminated by an else operator that changes the control context to the else block.
464 pushControlStack(ControlContext::Type::ifThen,imm.resultType,endBlock,endPHI,elseBlock);
465
466 // Push a branch target for the if end.
467 pushBranchTarget(imm.resultType,endBlock,endPHI);
468
469 }
470 void else_(NoImm imm)
471 {
472 WAVM_ASSERT_THROW(controlStack.size());
473 ControlContext& currentContext = controlStack.back();
474
475 if(currentContext.isReachable)
476 {
477 // If the control context expects a result, take it from the operand stack and add it to the
478 // control context's end PHI.
479 if(currentContext.resultType != ResultType::none)
480 {
481 llvm::Value* result = pop();
482 currentContext.endPHI->addIncoming(result,irBuilder.GetInsertBlock());
483 }
484
485 // Branch to the control context's end.
486 irBuilder.CreateBr(currentContext.endBlock);
487 }
488 WAVM_ASSERT_THROW(stack.size() == currentContext.outerStackSize);
489
490 // Switch the IR emitter to the else block.
491 WAVM_ASSERT_THROW(currentContext.elseBlock);
492 WAVM_ASSERT_THROW(currentContext.type == ControlContext::Type::ifThen);
493 currentContext.elseBlock->moveAfter(irBuilder.GetInsertBlock());
494 irBuilder.SetInsertPoint(currentContext.elseBlock);
495
496 // Change the top of the control stack to an else clause.
497 currentContext.type = ControlContext::Type::ifElse;
498 currentContext.isReachable = currentContext.isElseReachable;
499 currentContext.elseBlock = nullptr;
500 }
501 void end(NoImm)
502 {
503 WAVM_ASSERT_THROW(controlStack.size());
504 ControlContext& currentContext = controlStack.back();
505
506 if(currentContext.isReachable)
507 {
508 // If the control context yields a result, take the top of the operand stack and
509 // add it to the control context's end PHI.
510 if(currentContext.resultType != ResultType::none)
511 {
512 llvm::Value* result = pop();
513 currentContext.endPHI->addIncoming(result,irBuilder.GetInsertBlock());
514 }
515
516 // Branch to the control context's end.
517 irBuilder.CreateBr(currentContext.endBlock);
518 }
519 WAVM_ASSERT_THROW(stack.size() == currentContext.outerStackSize);
520
521 if(currentContext.elseBlock)
522 {
523 // If this is the end of an if without an else clause, create a dummy else clause.
524 currentContext.elseBlock->moveAfter(irBuilder.GetInsertBlock());
525 irBuilder.SetInsertPoint(currentContext.elseBlock);
526 irBuilder.CreateBr(currentContext.endBlock);
527 }
528
529 // Switch the IR emitter to the end block.
530 currentContext.endBlock->moveAfter(irBuilder.GetInsertBlock());
531 irBuilder.SetInsertPoint(currentContext.endBlock);
532
533 if(currentContext.endPHI)
534 {
535 // If the control context yields a result, take the PHI that merges all the control flow
536 // to the end and push it onto the operand stack.
537 if(currentContext.endPHI->getNumIncomingValues()) { push(currentContext.endPHI); }
538 else
539 {
540 // If there weren't any incoming values for the end PHI, remove it and push a dummy value.
541 currentContext.endPHI->eraseFromParent();
542 WAVM_ASSERT_THROW(currentContext.resultType != ResultType::none);
543 push(typedZeroConstants[(Uptr)asValueType(currentContext.resultType)]);
544 }
545 }
546
547 // Pop and branch targets introduced by this control context.
548 WAVM_ASSERT_THROW(currentContext.outerBranchTargetStackSize <= branchTargetStack.size());
549 branchTargetStack.resize(currentContext.outerBranchTargetStackSize);
550
551 // Pop this control context.
552 controlStack.pop_back();
553 }

577 void br_if(BranchImm imm)
578 {
579 // Pop the condition from operand stack.
580 auto condition = pop();
581
582 BranchTarget& target = getBranchTargetByDepth(imm.targetDepth);
583 if(target.argumentType != ResultType::none)
584 {
585 // Use the stack top as the branch argument (don't pop it) and add it to the target phi's incoming values.
586 llvm::Value* argument = getTopValue();
587 target.phi->addIncoming(argument,irBuilder.GetInsertBlock());
588 }
589
590 // Create a new basic block for the case where the branch is not taken.
591 auto falseBlock = llvm::BasicBlock::Create(context,"br_ifElse",llvmFunction);
592
593 // Emit a conditional branch to either the falseBlock or the target block.
594 irBuilder.CreateCondBr(coerceI32ToBool(condition),target.block,falseBlock);
595
596 // Resume emitting instructions in the falseBlock.
597 irBuilder.SetInsertPoint(falseBlock);
598 }

600 void br(BranchImm imm)
601 {
602 BranchTarget& target = getBranchTargetByDepth(imm.targetDepth);
603 if(target.argumentType != ResultType::none)
604 {
605 // Pop the branch argument from the stack and add it to the target phi's incoming values.
606 llvm::Value* argument = pop();
607 target.phi->addIncoming(argument,irBuilder.GetInsertBlock());
608 }
609
610 // Branch to the target block.
611 irBuilder.CreateBr(target.block);
612
613 enterUnreachable();
614 }
615 void br_table(BranchTableImm imm)
616 {
617 // Pop the table index from the operand stack.
618 auto index = pop();
619
620 // Look up the default branch target, and assume its argument type applies to all targets.
621 // (this is guaranteed by the validator)
622 BranchTarget& defaultTarget = getBranchTargetByDepth(imm.defaultTargetDepth);
623 const ResultType argumentType = defaultTarget.argumentType;
624 llvm::Value* argument = nullptr;
625 if(argumentType != ResultType::none)
626 {
627 // Pop the branch argument from the stack and add it to the default target phi's incoming values.
628 argument = pop();
629 defaultTarget.phi->addIncoming(argument,irBuilder.GetInsertBlock());
630 }
631
632 // Create a LLVM switch instruction.
633 WAVM_ASSERT_THROW(imm.branchTableIndex < functionDef.branchTables.size());
634 const std::vector<U32>& targetDepths = functionDef.branchTables[imm.branchTableIndex];
635 auto llvmSwitch = irBuilder.CreateSwitch(index,defaultTarget.block,(unsigned int)targetDepths.size());
636
637 for(Uptr targetIndex = 0;targetIndex < targetDepths.size();++targetIndex)
638 {
639 BranchTarget& target = getBranchTargetByDepth(targetDepths[targetIndex]);
640
641 // Add this target to the switch instruction.
642 llvmSwitch->addCase(emitLiteral((U32)targetIndex),target.block);
643
644 if(argumentType != ResultType::none)
645 {
646 // If this is the first case in the table for this branch target, add the branch argument to
647 // the target phi's incoming values.
648 target.phi->addIncoming(argument,irBuilder.GetInsertBlock());
649 }
650 }
651
652 enterUnreachable();
653 }
654 void return_(NoImm)
655 {
656 if(functionType->ret != ResultType::none)
657 {
658 // Pop the return value from the stack and add it to the return phi's incoming values.
659 llvm::Value* result = pop();
660 controlStack[0].endPHI->addIncoming(result,irBuilder.GetInsertBlock());
661 }
662
663 // Branch to the return block.
664 irBuilder.CreateBr(controlStack[0].endBlock);
665
666 enterUnreachable();
667 }

696 void call(CallImm imm)
697 {
698 // Map the callee function index to either an imported function pointer or a function in this module.
699 llvm::Value* callee;
700 const FunctionType* calleeType;
701 bool isExit = false;
702 if(imm.functionIndex < moduleContext.importedFunctionOffsets.size())
703 {
704 calleeType = module.types[module.functions.imports[imm.functionIndex].type.index];
705 llvm::Value* ic = irBuilder.CreateLoad( emitLiteralPointer((void*)(OFFSET_OF_FIRST_INTRINSIC-moduleContext.importedFunctionOffsets[imm.functionIndex]*8), llvmI64Type->getPointerTo(256)) );
706 callee = irBuilder.CreateIntToPtr(ic, asLLVMType(calleeType)->getPointerTo());
707 isExit = module.functions.imports[imm.functionIndex].moduleName == "env" && module.functions.imports[imm.functionIndex].exportName == "sysio_exit";
708 }
709 else
710 {
711 const Uptr calleeIndex = imm.functionIndex - moduleContext.importedFunctionOffsets.size();
712 callee = moduleContext.functionDefs[calleeIndex];
713 calleeType = module.types[module.functions.defs[calleeIndex].type.index];
714 }
715
716 // Pop the call arguments from the operand stack.
717 auto llvmArgs = (llvm::Value**)alloca(sizeof(llvm::Value*) * calleeType->parameters.size());
718 popMultiple(llvmArgs,calleeType->parameters.size());
719
720 // Call the function.
721 auto result = createCall(callee,llvm::ArrayRef<llvm::Value*>(llvmArgs,calleeType->parameters.size()));
722 if(isExit) {
723 irBuilder.CreateUnreachable();
724 enterUnreachable();
725 }
726
727 // Push the result on the operand stack.
728 if(calleeType->ret != ResultType::none) { push(result); }
729 }
730 void call_indirect(CallIndirectImm imm)
731 {
732 WAVM_ASSERT_THROW(imm.type.index < module.types.size());
733
734 auto calleeType = module.types[imm.type.index];
735 auto functionPointerType = asLLVMType(calleeType)->getPointerTo();
736
737 // Compile the function index.
738 auto tableElementIndex = pop();
739
740 // Compile the call arguments.
741 auto llvmArgs = (llvm::Value**)alloca(sizeof(llvm::Value*) * calleeType->parameters.size());
742 popMultiple(llvmArgs,calleeType->parameters.size());
743
744 // Zero extend the function index to the pointer size.
745 auto functionIndexZExt = irBuilder.CreateZExt(tableElementIndex, llvmI64Type);
746
747 // If the function index is larger than the function table size, trap.
748 emitConditionalTrapIntrinsic(
749 irBuilder.CreateICmpUGE(functionIndexZExt,moduleContext.defaultTableMaxElementIndex),
750 "eosvmoc_internal.indirect_call_oob",FunctionType::get(),{});
751
752 // Load the type for this table entry.
753 auto tablePointer = CreateInBoundsGEPWAR(irBuilder, moduleContext.defaultTablePointer, emitLiteral(0), emitLiteral(0));
754 auto functionTypePointerPointer = CreateInBoundsGEPWAR(irBuilder, tablePointer, functionIndexZExt, emitLiteral((U32)0));
755 auto functionTypePointer = irBuilder.CreateLoad(functionTypePointerPointer);
756 auto llvmCalleeType = emitLiteralPointer(calleeType,llvmI8PtrType);
757
758 // If the function type doesn't match, trap.
759 emitConditionalTrapIntrinsic(
760 irBuilder.CreateICmpNE(llvmCalleeType,functionTypePointer),
761 "eosvmoc_internal.indirect_call_mismatch",
762 FunctionType::get(),{}
763 );
764
765 //If the WASM only contains table elements to function definitions internal to the wasm, we can take a
766 // simple and approach
767 if(moduleContext.tableOnlyHasDefinedFuncs) {
768 auto functionPointerPointer = CreateInBoundsGEPWAR(irBuilder, tablePointer, functionIndexZExt, emitLiteral((U32)1));
769 auto functionInfo = irBuilder.CreateLoad(functionPointerPointer); //offset of code
770 llvm::Value* running_code_start = irBuilder.CreateLoad(emitLiteralPointer((void*)OFFSET_OF_CONTROL_BLOCK_MEMBER(running_code_base), llvmI64Type->getPointerTo(256)));
771 llvm::Value* offset_from_start = irBuilder.CreateAdd(running_code_start, functionInfo);
772 llvm::Value* ptr_cast = irBuilder.CreateIntToPtr(offset_from_start, functionPointerType);
773 auto result = createCall(ptr_cast,llvm::ArrayRef<llvm::Value*>(llvmArgs,calleeType->parameters.size()));
774
775 // Push the result on the operand stack.
776 if(calleeType->ret != ResultType::none) { push(result); }
777 }
778 else {
779 auto functionPointerPointer = CreateInBoundsGEPWAR(irBuilder, tablePointer, functionIndexZExt, emitLiteral((U32)1));
780 auto functionInfo = irBuilder.CreateLoad(functionPointerPointer); //offset of code
781
782 auto is_intrnsic = irBuilder.CreateICmpSLT(functionInfo, typedZeroConstants[(Uptr)ValueType::i64]);
783
784 llvm::BasicBlock* is_intrinsic_block = llvm::BasicBlock::Create(context, "isintrinsic", llvmFunction);
785 llvm::BasicBlock* is_code_offset_block = llvm::BasicBlock::Create(context, "isoffset");
786 llvm::BasicBlock* continuation_block = llvm::BasicBlock::Create(context, "cont");
787
788 irBuilder.CreateCondBr(is_intrnsic, is_intrinsic_block, is_code_offset_block, moduleContext.likelyFalseBranchWeights);
789
790 irBuilder.SetInsertPoint(is_intrinsic_block);
791 llvm::Value* intrinsic_start = emitLiteral((I64)OFFSET_OF_FIRST_INTRINSIC);
792 llvm::Value* intrinsic_offset = irBuilder.CreateAdd(intrinsic_start, functionInfo);
793 llvm::Value* intrinsic_ptr = irBuilder.CreateLoad(irBuilder.CreateIntToPtr(intrinsic_offset, llvmI64Type->getPointerTo(256)));
794 irBuilder.CreateBr(continuation_block);
795
796 llvmFunction->getBasicBlockList().push_back(is_code_offset_block);
797 irBuilder.SetInsertPoint(is_code_offset_block);
798 llvm::Value* running_code_start = irBuilder.CreateLoad(emitLiteralPointer((void*)OFFSET_OF_CONTROL_BLOCK_MEMBER(running_code_base), llvmI64Type->getPointerTo(256)));
799 llvm::Value* offset_from_start = irBuilder.CreateAdd(running_code_start, functionInfo);
800 irBuilder.CreateBr(continuation_block);
801
802 llvmFunction->getBasicBlockList().push_back(continuation_block);
803 irBuilder.SetInsertPoint(continuation_block);
804
805 llvm::PHINode* PN = irBuilder.CreatePHI(llvmI64Type, 2, "indirecttypephi");
806 PN->addIncoming(intrinsic_ptr, is_intrinsic_block);
807 PN->addIncoming(offset_from_start, is_code_offset_block);
808
809 llvm::Value* ptr_cast = irBuilder.CreateIntToPtr(PN, functionPointerType);
810 auto result = createCall(ptr_cast,llvm::ArrayRef<llvm::Value*>(llvmArgs,calleeType->parameters.size()));
811
812 // Push the result on the operand stack.
813 if(calleeType->ret != ResultType::none) { push(result); }
814 }
815 }

973 llvm::Value* emitSRem(ValueType type,llvm::Value* left,llvm::Value* right)
974 {
975 // Trap if the dividend is zero.
976 trapDivideByZero(type,right);
977
978 // LLVM's srem has undefined behavior where WebAssembly's rem_s defines that it should not trap if the corresponding
979 // division would overflow a signed integer. To avoid this case, we just branch around the srem if the INT_MAX%-1 case
980 // that overflows is detected.
981 auto preOverflowBlock = irBuilder.GetInsertBlock();
982 auto noOverflowBlock = llvm::BasicBlock::Create(context,"sremNoOverflow",llvmFunction);
983 auto endBlock = llvm::BasicBlock::Create(context,"sremEnd",llvmFunction);
984 auto noOverflow = irBuilder.CreateOr(
985 irBuilder.CreateICmpNE(left,type == ValueType::i32 ? emitLiteral((U32)INT32_MIN) : emitLiteral((U64)INT64_MIN)),
986 irBuilder.CreateICmpNE(right,type == ValueType::i32 ? emitLiteral((U32)-1) : emitLiteral((U64)-1))
987 );
988 irBuilder.CreateCondBr(noOverflow,noOverflowBlock,endBlock,moduleContext.likelyTrueBranchWeights);
989
990 irBuilder.SetInsertPoint(noOverflowBlock);
991 auto noOverflowValue = irBuilder.CreateSRem(left,right);
992 irBuilder.CreateBr(endBlock);
993
994 irBuilder.SetInsertPoint(endBlock);
995 auto phi = irBuilder.CreatePHI(asLLVMType(type),2);
996 phi->addIncoming(typedZeroConstants[(Uptr)type],preOverflowBlock);
997 phi->addIncoming(noOverflowValue,noOverflowBlock);
998 return phi;
999 }

1004 // grather than the bit count of the operands. This matches x86's native shift instructions, but explicitly mask

1006 auto bitsMinusOne = irBuilder.CreateZExt(emitLiteral((U8)(getTypeBitWidth(type) - 1)),asLLVMType(type));
1007 return irBuilder.CreateAnd(shiftCount,bitsMinusOne);
1008 }

41COLOR_ENV_VAR = 'GTEST_COLOR'

46def SetEnvVar(env_var, value):
47 """Sets the env variable to 'value'; unsets it when 'value' is None."""
48
49 if value is not None:
50 os.environ[env_var] = value
51 elif env_var in os.environ:
52 del os.environ[env_var]

55def UsesColor(term, color_env_var, color_flag):
56 """Runs gtest_color_test_ and returns its exit code."""
57
58 SetEnvVar('TERM', term)
59 SetEnvVar(COLOR_ENV_VAR, color_env_var)
60
61 if color_flag is None:
62 args = []
63 else:
64 args = ['--%s=%s' % (COLOR_FLAG, color_flag)]
65 p = gtest_test_utils.Subprocess([COMMAND] + args)
66 return not p.exited or p.exit_code

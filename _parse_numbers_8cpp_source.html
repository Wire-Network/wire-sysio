53// Parses an optional + or - sign and returns true if a - sign was parsed.
54// If either a + or - sign is parsed, nextChar is advanced past it.
55static bool parseSign(const char*& nextChar)
56{
57 if(*nextChar == '-') { ++nextChar; return true; }
58 else if(*nextChar == '+') { ++nextChar; }
59 return false;
60}

62// Parses an unsigned integer from hexits, starting with "0x", and advancing nextChar past the parsed hexits.
63// be called for input that's already been accepted by the lexer as a hexadecimal integer.
64static U64 parseHexUnsignedInt(const char*& nextChar,ParseState& state,U64 maxValue)
65{
66 const char* firstHexit = nextChar;
67 WAVM_ASSERT_THROW(nextChar[0] == '0' && (nextChar[1] == 'x' || nextChar[1] == 'X'));
68 nextChar += 2;
69
70 U64 result = 0;
71 U8 hexit = 0;
72 while(true)
73 {
74 if(*nextChar == '_') { ++nextChar; continue; }
75 if(!tryParseHexit(nextChar,hexit)) { break; }
76 if(result > (maxValue - hexit) / 16)
77 {
78 parseErrorf(state,firstHexit,"integer literal is too large");
79 result = maxValue;
80 while(tryParseHexit(nextChar,hexit)) {};
81 break;
82 }
83 WAVM_ASSERT_THROW(result * 16 + hexit >= result);
84 result = result * 16 + hexit;
85 }
86 return result;
87}

89// Parses an unsigned integer from digits, advancing nextChar past the parsed digits.
90// Assumes it will only be called for input that's already been accepted by the lexer as a decimal integer.
91static U64 parseDecimalUnsignedInt(const char*& nextChar,ParseState& state,U64 maxValue,const char* context)
92{
93 U64 result = 0;
94 const char* firstDigit = nextChar;
95 while(true)
96 {
97 if(*nextChar == '_') { ++nextChar; continue; }
98 if(*nextChar < '0' || *nextChar > '9') { break; }
99
100 const U8 digit = *nextChar - '0';
101 ++nextChar;
102
103 if(result > U64(maxValue - digit) / 10)
104 {
105 parseErrorf(state,firstDigit,"%s is too large",context);
106 result = maxValue;
107 while((*nextChar >= '0' && *nextChar <= '9') || *nextChar == '_') { ++nextChar; };
108 break;
109 }
110 WAVM_ASSERT_THROW(result * 10 + digit >= result);
111 result = result * 10 + digit;
112 };
113 return result;
114}

116// Parses a floating-point NaN, advancing nextChar past the parsed characters.
117// Assumes it will only be called for input that's already been accepted by the lexer as a literal NaN.
118template<typename Float>
119Float parseNaN(const char*& nextChar,ParseState& state)
120{
121 typedef typename Floats::FloatComponents<Float> FloatComponents;
122 FloatComponents resultComponents;
123 resultComponents.bits.sign = parseSign(nextChar) ? 1 : 0;
124 resultComponents.bits.exponent = FloatComponents::maxExponentBits;
125
126 WAVM_ASSERT_THROW(nextChar[0] == 'n'
127 && nextChar[1] == 'a'
128 && nextChar[2] == 'n');
129 nextChar += 3;
130
131 if(*nextChar == ':')
132 {
133 ++nextChar;
134
135 const U64 significandBits = parseHexUnsignedInt(nextChar,state,FloatComponents::maxSignificand);
136 resultComponents.bits.significand = typename FloatComponents::Bits(significandBits);
137 }
138 else
139 {
140 // If the NaN's significand isn't specified, just set the top bit.
141 resultComponents.bits.significand = typename FloatComponents::Bits(1) << (FloatComponents::numSignificandBits-1);
142 }
143
144 return resultComponents.value;
145}

148// Parses a floating-point infinity. Does not advance nextChar.
149// Assumes it will only be called for input that's already been accepted by the lexer as a literal infinity.
150template<typename Float>
151Float parseInfinity(const char* nextChar)
152{
153 // Floating point infinite is represented by max exponent with a zero significand.
154 typedef typename Floats::FloatComponents<Float> FloatComponents;
155 FloatComponents resultComponents;
156 resultComponents.bits.sign = parseSign(nextChar) ? 1 : 0;
157 resultComponents.bits.exponent = FloatComponents::maxExponentBits;
158 resultComponents.bits.significand = 0;
159 return resultComponents.value;
160}

162// Parses a decimal floating point literal, advancing nextChar past the parsed characters.
163// Assumes it will only be called for input that's already been accepted by the lexer as a decimal float literal.
164template<typename Float>
165Float parseFloat(const char*& nextChar,ParseState& state)
166{
167 // Scan the token's characters for underscores, and make a copy of it without the underscores for strtod.
168 const char* firstChar = nextChar;
169 std::string noUnderscoreString;
170 bool hasUnderscores = false;
171 while(true)
172 {
173 // Determine whether the next character is still part of the number.
174 bool isNumericChar = false;
175 switch(*nextChar)
176 {
177 case '0': case '1': case '2': case '3': case '4': case '5': case '6': case '7': case '8': case '9':
178 case 'a': case 'b': case 'c': case 'd': case 'e': case 'f':
179 case 'A': case 'B': case 'C': case 'D': case 'E': case 'F':
180 case '+': case '-': case 'x': case 'X': case '.': case 'p': case 'P': case '_':
181 isNumericChar = true;
182 break;
183 };
184 if(!isNumericChar) { break; }
185
186 if(*nextChar == '_' && !hasUnderscores)
187 {
188 // If this is the first underscore encountered, copy the preceding characters of the number to a std::string.
189 noUnderscoreString = std::string(firstChar,nextChar);
190 hasUnderscores = true;
191 }
192 else if(*nextChar != '_' && hasUnderscores)
193 {
194 // If an underscore has previously been encountered, copy non-underscore characters to that string.
195 noUnderscoreString += *nextChar;
196 }
197
198 ++nextChar;
199 };
200
201 // Pass the underscore-free string to parseNonSpecialF64 instead of the original input string.
202 if(hasUnderscores) { firstChar = noUnderscoreString.c_str(); }
203
204 // Use David Gay's strtod to parse a floating point number.
205 char* endChar = nullptr;
206 F64 f64 = parseNonSpecialF64(firstChar,&endChar);
207 if(endChar == firstChar)
208 {
209 Errors::fatalf("strtod failed to parse number accepted by lexer");
210 }
211 if(Float(f64) < std::numeric_limits<Float>::lowest() || Float(f64) > std::numeric_limits<Float>::max())
212 {
213 parseErrorf(state,firstChar,"float literal is too large");
214 }
215
216 return (Float)f64;
217}

219// Tries to parse an numeric literal token as an integer, advancing state.nextToken.
220// Returns true if it matched a token.
221template<typename UnsignedInt>
222bool tryParseInt(ParseState& state,UnsignedInt& outUnsignedInt,I64 minSignedValue,U64 maxUnsignedValue)
223{
224 bool isNegative = false;
225 U64 u64 = 0;
226
227 const char* nextChar = state.string + state.nextToken->begin;
228 switch(state.nextToken->type)
229 {
230 case t_decimalInt:
231 isNegative = parseSign(nextChar);
232 u64 = parseDecimalUnsignedInt(nextChar,state,isNegative ? U64(-minSignedValue) : maxUnsignedValue,"int literal");
233 break;
234 case t_hexInt:
235 isNegative = parseSign(nextChar);
236 u64 = parseHexUnsignedInt(nextChar,state,isNegative ? U64(-minSignedValue) : maxUnsignedValue);
237 break;
238 default:
239 return false;
240 };
241
242 outUnsignedInt = isNegative ? UnsignedInt(-I64(u64)) : UnsignedInt(u64);
243
244 ++state.nextToken;
245 WAVM_ASSERT_THROW(nextChar <= state.string + state.nextToken->begin);
246
247 return true;
248}

250// Tries to parse a numeric literal literal token as a float, advancing state.nextToken.
251// Returns true if it matched a token.
252template<typename Float>
253bool tryParseFloat(ParseState& state,Float& outFloat)
254{
255 const char* nextChar = state.string + state.nextToken->begin;
256 switch(state.nextToken->type)
257 {
258 case t_decimalInt:
259 case t_decimalFloat: outFloat = parseFloat<Float>(nextChar,state); break;
260 case t_hexInt:
261 case t_hexFloat: outFloat = parseFloat<Float>(nextChar,state); break;
262 case t_floatNaN: outFloat = parseNaN<Float>(nextChar,state); break;
263 case t_floatInf: outFloat = parseInfinity<Float>(nextChar); break;
264 default:
265 parseErrorf(state,state.nextToken,"expected float literal");
266 return false;
267 };
268
269 ++state.nextToken;
270 WAVM_ASSERT_THROW(nextChar <= state.string + state.nextToken->begin);
271
272 return true;
273}

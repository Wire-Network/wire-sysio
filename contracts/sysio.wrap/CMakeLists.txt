if(BUILD_SYSTEM_CONTRACTS)
  # When building from source, copy pre-built contracts only if they don't exist (bootstrap)
  if(NOT EXISTS ${CMAKE_CURRENT_BINARY_DIR}/sysio.wrap.wasm)
    configure_file(${CMAKE_CURRENT_SOURCE_DIR}/sysio.wrap.wasm ${CMAKE_CURRENT_BINARY_DIR}/sysio.wrap.wasm COPYONLY)
  endif()
  if(NOT EXISTS ${CMAKE_CURRENT_BINARY_DIR}/sysio.wrap.abi)
    configure_file(${CMAKE_CURRENT_SOURCE_DIR}/sysio.wrap.abi ${CMAKE_CURRENT_BINARY_DIR}/sysio.wrap.abi COPYONLY)
  endif()
  add_contract(sysio.wrap sysio.wrap sysio.wrap.cpp)
else()
  # When not building from source, always copy pre-built contracts (ensures fresh copy on branch switch)
  configure_file(${CMAKE_CURRENT_SOURCE_DIR}/sysio.wrap.wasm ${CMAKE_CURRENT_BINARY_DIR}/sysio.wrap.wasm COPYONLY)
  configure_file(${CMAKE_CURRENT_SOURCE_DIR}/sysio.wrap.abi ${CMAKE_CURRENT_BINARY_DIR}/sysio.wrap.abi COPYONLY)
endif()

21template<typename C>
22class TestInvokerAsMethod : public ITestInvoker {
23 void (C::*m_testAsMethod)();
24public:
25 TestInvokerAsMethod( void (C::*testAsMethod)() ) noexcept : m_testAsMethod( testAsMethod ) {}
26
27 void invoke() const override {
28 C obj;
29 (obj.*m_testAsMethod)();
30 }
31};

33auto makeTestInvoker( void(*testAsFunction)() ) noexcept -> ITestInvoker*;

35template<typename C>
36auto makeTestInvoker( void (C::*testAsMethod)() ) noexcept -> ITestInvoker* {

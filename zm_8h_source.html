85{
86 std::ios_base::fmtflags keep = is.flags();
87 size_t pos = 0;
88 char c;
89 while (is >> c) {
90 if (('0' <= c && c <= '9') /* digits */
91 || (pos == 1 && (str[0] == '0' && c == 'x')) /* 0x.. */
92 || ('a' <= c && c <= 'f') /* lowercase hex */
93 || ('A' <= c && c <= 'F') /* uppercase hex */
94 || (allowNegative && pos == 0 && c == '-')) { /* -digits */

96 if (pos == 0) {
97 is >> std::noskipws;
98 }
99 pos++;
100 } else {

103 }
